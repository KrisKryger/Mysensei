CREATE PROCEDURE [dbo].GetRandomCourses
	@coursesToReturn int
AS
	SET NOCOUNT ON;
	SELECT TOP(@coursesToReturn) *
	FROM Courses
	WHERE CourseStatusId = 1
	OR CourseStatusId = 2
	ORDER BY NEWID()
